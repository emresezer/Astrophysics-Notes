\subsection{Proton-Proton Döngüsü}

\begin{align}
{}^{1}\mathrm{H} + {}^{1}\mathrm{H} &\rightarrow {}^{2}\mathrm{H} + e^{+} + \nu_{e} \\
{}^{2}\mathrm{H} + {}^{1}\mathrm{H}  &\rightarrow {}^{3}\mathrm{He} + \gamma \\
{}^{3}\mathrm{He} + {}^{3}\mathrm{He} &\rightarrow {}^{4}\mathrm{He} + {}^{1}\mathrm{H} + {}^{1}\mathrm{H} + 26.7MeV
\end{align}


\subsection{C-N-O Döngüsü}

\begin{aligned} 
^{12}\mathrm{C} + p &\rightarrow \, ^{13}\mathrm{N} + \gamma \\ ^{13}\mathrm{N} &\rightarrow \, ^{13}\mathrm{C} + e^{+} + \nu_{e} \\ ^{13}\mathrm{C} + p &\rightarrow \, ^{14}\mathrm{N} + \gamma \\ ^{14}\mathrm{N} + p &\rightarrow \, ^{15}\mathrm{O} + \gamma \\ ^{15}\mathrm{O} &\rightarrow \, ^{15}\mathrm{N} + e^{+} + \nu_{e} \\ ^{15}\mathrm{N} + p &\rightarrow \, ^{12}\mathrm{C} + {}^{4}\mathrm{He} + 25MeV/4 P
\end{aligned}
