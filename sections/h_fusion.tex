\subsection{Proton-Proton Döngüsü}

\begin{align}
{}^{1}\mathrm{H} + {}^{1}\mathrm{H} &\rightarrow {}^{2}\mathrm{H} + e^{+} + \nu_e \\
{}^{2}\mathrm{H} + {}^{1}\mathrm{H} &\rightarrow {}^{3}\mathrm{He} + \gamma \\
{}^{3}\mathrm{He} + {}^{3}\mathrm{He} &\rightarrow {}^{4}\mathrm{He} + {}^{1}\mathrm{H} + {}^{1}\mathrm{H} + 26.7\,\mathrm{MeV}
\end{align}

\subsection{C-N-O Döngüsü}

\begin{align}
{}^{12}\mathrm{C} + {}^{1}\mathrm{H} &\rightarrow {}^{13}\mathrm{N} + \gamma \\
{}^{13}\mathrm{N} &\rightarrow {}^{13}\mathrm{C} + e^{+} + \nu_e \\
{}^{13}\mathrm{C} + {}^{1}\mathrm{H} &\rightarrow {}^{14}\mathrm{N} + \gamma \\
{}^{14}\mathrm{N} + {}^{1}\mathrm{H} &\rightarrow {}^{15}\mathrm{O} + \gamma \\
{}^{15}\mathrm{O} &\rightarrow {}^{15}\mathrm{N} + e^{+} + \nu_e \\
{}^{15}\mathrm{N} + p &\rightarrow {}^{12}\mathrm{C} + {}^{4}\mathrm{He} + \frac{25\,\mathrm{MeV}}{4\,\mathrm{P}}
\end{align}
